Extract Timeperiods 
#+begin_src lisp 
(defparameter tps (loop for (x tp) in moi::*bigdata* collect tp))
#+end_src

Calulate Histogram
#+begin_src lisp 
     (defparameter hist (histogram tps :steps 0.02))
#+end_src

#+RESULTS:
: HIST

Plot textually 
#+begin_src lisp
  (eazy-gnuplot:with-plots (*standard-output* :debug t)
    (eazy-gnuplot:gp-setup :terminal "png"
			   :output (merge-pathnames "histogram.png" moi::*img-path*))
    (eazy-gnuplot:plot (lambda ()
			 (let ((count 0))
			   (map nil (lambda (v)
				      (format t "~,3f ~,4f~%" (incf count 0.02) v))
				hist)))
		       :with '(:lines)))
#+end_src
